export AWS_PAGER=cat

# Crear key pair
aws ec2 create-key-pair --key-name practica001-key --query 'KeyMaterial' --output text --region us-east-1 > practica001-key.pem
chmod 400 practica001-key.pem

# Crear grupos de seguridad
aws ec2 create-security-group --group-name practica001-sg --description "security para la practica001" --region us-east-1
aws ec2 authorize-security-group-ingress --group-id sg-0ecd2204d90c207a2 --protocol tcp --port 22 --cidr 0.0.0.0/0 --region us-east-1
aws ec2 authorize-security-group-ingress --group-id sg-0ecd2204d90c207a2 --protocol tcp --port 8080 --cidr 0.0.0.0/0 --region us-east-1

# la siguiente linea es para ver que se haya creado el sg
# aws ec2 describe-security-groups --group-ids <sg-id> --query 'SecurityGroups[].IpPermissions'


# Crear una instancia EC2
# El ID del AMI lo puedes obtener en la consola de AWS.
aws ec2 run-instances \
    --image-id ami-0532be01f26a3de55 \
    --count 1 \
    --instance-type t3.micro \
    --key-name practica001-key \
    --security-groups practica001-sg \
    --region us-east-1 \
    --tag-specifications 'ResourceType=instance,Tags=[{Key=Name,Value=practica1-ec2}]'

# Ver IP pública de la instancia EC2  PARA PODER CONECTARNOS
aws ec2 describe-instances \
  --filters "Name=tag:Name,Values=practica001-ec2" \
  --query 'Reservations[].Instances[].PublicIpAddress' \
  --output text \
  --region us-east-1
